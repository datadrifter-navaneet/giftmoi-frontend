from flask import Flask, request, jsonify from flask_cors import CORS import firebase_admin from firebase_admin import credentials, firestore import qrcode import io import base64 import uuid import secrets import re from datetime import datetime from decimal import Decimal, ROUND_HALF_UP import os import json import urllib.parse app = Flask(__name__) CORS(app) # ── Firebase init ── # On Railway: reads from FIREBASE_CREDENTIALS environment variable # On local machine: reads from serviceAccount.json file firebase_creds = os.environ.get('FIREBASE_CREDENTIALS') if firebase_creds: cred = credentials.Certificate(json.loads(firebase_creds)) else: cred = credentials.Certificate('serviceAccount.json') firebase_admin.initialize_app(cred) db = firestore.client() # ───────────────────────────────────────── # Helper: sanitize a name for UPI fields # Strips everything except alphanumeric + spaces, then trims to max_len. # ───────────────────────────────────────── def sanitize_name(raw: str, max_len: int = 50) -> str: return re.sub(r'[^a-zA-Z0-9 ]', '', raw).strip()[:max_len] # ───────────────────────────────────────── # Helper: format amount as a clean decimal string # Uses Decimal to avoid IEEE-754 float precision issues. # Returns "501" for whole numbers, "501.50" for fractional. # ───────────────────────────────────────── def format_amount(raw_amount) -> str: d = Decimal(str(raw_amount)).quantize(Decimal('0.01'), rounding=ROUND_HALF_UP) if d == d.to_integral_value(): return str(int(d)) return str(d) # ───────────────────────────────────────── # POST /event # Host creates a new event. # Body: { "host_name": "Rahul", "upi_id": "rahul@oksbi", "occasion": "Birthday" } # Returns: { "event_id": "...", "qr_code": "base64 image", "qr_url": "..." } # ───────────────────────────────────────── @app.route('/event', methods=['POST']) def create_event(): data = request.get_json() # FIX: Sanitize host_name at creation time so Firestore never stores # special characters that would later break UPI URL construction. host_name = sanitize_name(data.get('host_name', ''), max_len=50) upi_id = data.get('upi_id', '').strip() occasion = data.get('occasion', '').strip()[:100] # FIX: Validate UPI ID format — must contain exactly one '@' if not host_name or not upi_id or upi_id.count('@') != 1: return jsonify({'error': 'host_name and a valid upi_id (format: handle@bank) are required'}), 400 # FIX: Validate UPI ID contains only safe characters # UPI IDs are alphanumeric + '.', '-', '_' before '@'; alphanumeric after. upi_local, upi_psp = upi_id.split('@') if not re.match(r'^[a-zA-Z0-9.\-_]+$', upi_local) or not re.match(r'^[a-zA-Z0-9]+$', upi_psp): return jsonify({'error': 'Invalid UPI ID format'}), 400 event_id = str(uuid.uuid4())[:8].upper() db.collection('events').document(event_id).set({ 'host_name': host_name, # stored clean 'upi_id': upi_id, 'occasion': occasion, 'created_at': datetime.utcnow().isoformat(), 'event_id': event_id }) base_url = os.environ.get('FRONTEND_URL', 'https://giftmoi.vercel.app') attendee_url = ( f"{base_url}?" f"upi={urllib.parse.quote(upi_id, safe='')}" f"&name={urllib.parse.quote(host_name, safe='')}" f"&occasion={urllib.parse.quote(occasion, safe='')}" f"&event_id={event_id}" ) qr = qrcode.QRCode( version=1, error_correction=qrcode.constants.ERROR_CORRECT_H, box_size=10, border=4 ) qr.add_data(attendee_url) qr.make(fit=True) img = qr.make_image(fill_color='#5C3A1E', back_color='white') buffer = io.BytesIO() img.save(buffer, format='PNG') qr_base64 = base64.b64encode(buffer.getvalue()).decode('utf-8') return jsonify({ 'event_id': event_id, 'qr_code': f'data:image/png;base64,{qr_base64}', 'qr_url': attendee_url, 'host_name': host_name, 'occasion': occasion }), 201 # ───────────────────────────────────────── # POST /gift # Attendee submits their gift entry. # Body: { "event_id": "...", "attendee_name": "...", "amount": 501, "wish": "..." } # Returns: { "success": true, "gift_id": "...", "upi_params": { pa, pn, am, tn, tr, cu } } # # IMPORTANT: Returns structured upi_params (not a pre-built URL string). # The frontend is responsible for encoding each field exactly once when # constructing intent:// and upi:// URLs. This eliminates the entire class # of double-encoding bugs that previously caused UPI ID corruption. # ───────────────────────────────────────── @app.route('/gift', methods=['POST']) def submit_gift(): data = request.get_json() event_id = data.get('event_id', '').strip() attendee_name = data.get('attendee_name', '').strip() amount = data.get('amount') wish = data.get('wish', '').strip() if not event_id or not attendee_name or not amount: return jsonify({'error': 'event_id, attendee_name and amount are required'}), 400 # Validate amount is a positive number try: amount_numeric = float(amount) if amount_numeric <= 0: raise ValueError except (ValueError, TypeError): return jsonify({'error': 'amount must be a positive number'}), 400 event_ref = db.collection('events').document(event_id).get() if not event_ref.exists: return jsonify({'error': 'Event not found'}), 404 event_data = event_ref.to_dict() gift_id = str(uuid.uuid4())[:8] db.collection('events').document(event_id).collection('gifts').document(gift_id).set({ 'gift_id': gift_id, 'attendee_name': attendee_name, 'amount': amount_numeric, 'wish': wish, 'timestamp': datetime.utcnow().isoformat(), 'status': 'attempted' }) # FIX: Use Decimal to avoid IEEE-754 float precision errors on amount. amount_str = format_amount(amount_numeric) # FIX: Transaction note — no spaces, no special characters. # Spaces inside tn caused %20 encoding issues that triggered payment failures # on certain PhonePe and GPay builds. Stripped to pure alphanumeric only. clean_name = re.sub(r'[^a-zA-Z0-9]', '', attendee_name)[:20] note = f"GiftMoi{clean_name}" # e.g. "GiftMoiRahulVerma" — max ~28 chars # FIX: Truly unique transaction reference using CSPRNG (secrets module), # NOT a truncated UUID. Truncating uuid4 to 12 chars collapses the # collision space from 2^122 to ~2^48, causing real-world tr collisions # across concurrent attendees. UPI apps silently report this as # "Transaction limit exceeded" rather than "duplicate transaction". # "GM" prefix + 30 random hex chars = 32 chars total, well within UPI's # 50-char tr limit, and astronomically unlikely to collide. tr = "GM" + secrets.token_hex(15).upper() # FIX: Defensive re-sanitize pn in case an older event was stored before # the sanitize-at-creation fix was deployed. clean_pn = sanitize_name(event_data['host_name'], max_len=50) # Return structured fields. Frontend encodes each value exactly once. # Do NOT return a pre-built upi:// URL string — that caused the # parse-then-re-encode pipeline that introduced double-encoding bugs. return jsonify({ 'success': True, 'gift_id': gift_id, 'upi_params': { 'pa': event_data['upi_id'], # raw UPI ID, safe (validated at /event) 'pn': clean_pn, # alphanumeric + spaces only 'am': amount_str, # e.g. "501" or "501.50" 'tn': note, # alphanumeric only, no spaces 'tr': tr, # CSPRNG-backed 32-char unique ref 'cu': 'INR' } }), 201 # ───────────────────────────────────────── # GET /ledger/ # Host fetches all gift entries for their event. # Returns: { "event": {...}, "gifts": [...], "total": 0000 } # ───────────────────────────────────────── @app.route('/ledger/', methods=['GET']) def get_ledger(event_id): event_ref = db.collection('events').document(event_id).get() if not event_ref.exists: return jsonify({'error': 'Event not found'}), 404 event_data = event_ref.to_dict() gifts_ref = db.collection('events').document(event_id).collection('gifts').stream() gifts = [] total = Decimal('0') for gift in gifts_ref: g = gift.to_dict() gifts.append(g) # FIX: Use Decimal for total accumulation to avoid float drift total += Decimal(str(g.get('amount', 0))) gifts.sort(key=lambda x: x.get('timestamp', ''), reverse=True) return jsonify({ 'event': event_data, 'gifts': gifts, 'total': float(total), 'gift_count': len(gifts) }), 200 # ───────────────────────────────────────── # Health check # ───────────────────────────────────────── @app.route('/', methods=['GET']) def health(): return jsonify({'status': 'GiftMoi backend is live 🎁'}), 200 if __name__ == '__main__': app.run(host='0.0.0.0', port=8080, debug=False)
